INSERT INTO user_coupons (id, user_id, coupon_type_id, acquired_at, expires_at)
SELECT
  seq_user_coupons.NEXTVAL,
  u.id,
  ct.id,
  CAST(CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Tokyo' AS TIMESTAMP),
  CAST(CURRENT_TIMESTAMP AT TIME ZONE 'Asia/Tokyo' AS TIMESTAMP)
    + NUMTODSINTERVAL(ct.validity_days, 'DAY')
FROM users u
JOIN coupon_types ct
  ON ct.id = (
    SELECT MIN(id)
    FROM coupon_types
    WHERE discount_rate = 5
      AND active_flag = 1
  )
WHERE u.email = 'ippan_saburo@test.co.jp';
